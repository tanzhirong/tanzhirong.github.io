Analysis of MNIST Data with EM Algorithm¶

In this project, we will work with the MNIST digits data of over 70 000 observations. We will code the EM algorithm directly instead of relying on a package.

We implement the expectation maximisation (EM) algorithm on this data for each class of digits (0,1,2,...,9). We will fine-tune our EM algorithm by varying the number of mixture models, M being assumed, where we will let M = 1; 3; 5.

For the 3 scenarios, where M = 1,3,5, we will try to learn a generative model for each digit class. Then, we will classify the test set data using Bayes’ rule with the estimated class mixture models and report the results for M=1,3,5.

"""
Define Class for EM
"""
class EM:
    """
    Create a new object for every digit c, c = 0, ... 9, 
    and every M =1,3,5
    """
    def __init__(self, c, M):
        self.c=c; self.M=M; self.components = 784
    
    
    """
    Define full alogrithm. Running this function will start 
    the EM algorithm.
    """
    def full_algorithm(self, threshold):
        #Find data that relates to label c
        #Subset this dataset
        idx_k = np.where(train_y == self.c)[0]
        self.train_x_k = train_x[idx_k,:]
        self.n = self.train_x_k.shape[0]

        """
        Next, create empty vectors to hold various values
        for the iterations
        """
        #W (n x M matrix, each is w_im)
        self.W = np.zeros((self.n,self.M))
        #bernoulli parameter p_jm (components x M matrix)
        self.P = np.zeros((self.components, self.M))
        #Pi_m (M vector)        
        self.pi = np.zeros((self.M))
        #Matrix of log-likelihood (pi_m * f_m)
        self.loglik = np.zeros((self.n, self.M))
        
        """
        Initiate initial values
        """
        self.overall_loglik = 0
        #Set initial difference to be big
        self.difference = 1000000000000000
        
        #Initialize value of W
        for i in range(self.n):
            #randomly choose a component's weight = 1, for all i
            self.W[i, np.random.choice(range(M), 1)[0]] = 1
        
        #START ITERATION
        self.count = 0
        
        """
        Begin the iteration. Continue as long as the diff
        is more than the threshold given in function
        """
        while self.difference >= threshold:
            """
            Estimate pi based on values of new W
            Estimate p_jm based on new values of W
            """
            self.count = self.count + 1
            
            #restimate pi
            for m in range(self.M):
                self.pi[m] = (1 + sum(self.W[:,m])) / (self.n + 2)
            
            #reestimate p_jm
            for j in range(self.components):
                for m in range(self.M):
                    bottom = sum(self.W[:,m])+2
                    top = sum(self.W[:,m] * self.train_x_k[:,j]) + 1
                    self.P[j,m] = top/bottom
            
            """
            COMPUTE new W
            """
            #Log-likelihood of each i under each m
            #calculated with new estiamtes of p_jm and pi
            for m in range(self.M):
                for i in range(self.n):
                    a = np.log(self.pi[m])
                    b = sum(self.train_x_k[i,:]*np.log(self.P[:,m])+\
                            (1-self.train_x_k[i,:])*
                            (np.log(1-self.P[:,m])))
                    self.loglik[i,m] = a + b
            for i in range(self.n):
                #Find largest one - L
                L = max(self.loglik[i,:])
                for m in range(self.M):
                    top = np.exp(self.loglik[i,m] - L)
                    bottom = sum([np.exp(self.loglik[i,cat] - L) 
                                  for cat in range(self.M)])
                    self.W[i,m] = top/bottom
            
            #KEEP the previous value before we erase it
            self.overall_loglik_2 = copy.deepcopy(self.overall_loglik)
            self.overall_loglik = sum(np.log(np.sum(np.exp(self.loglik),
                                                    axis=1)))
            
            #Find change in log-likelihood
            self.difference = self.overall_loglik - self.overall_loglik_2
            #If difference > threshold, loop will stop
            #Otherwise loop will continue
            
        #Report Statistics once loop ends
        print("M=",self.M, " K=", c, " #Times=",self.count,
             " Loglikelihood=", self.overall_loglik, " Diff=", 
              self.difference)
        
        return
    
    """
    Find loglik for the test set for a given m 
    and given k
    Out: predicted loglikelihood for test data, for each digit
    """
    def predicted_loglik(self, X_test):
        test_n = X_test.shape[0]
        self.test_results = np.zeros((test_n))
        for i in range(X_test.shape[0]):
            self.predict_loglik[i] = sum(np.exp([np.log(self.pi[m]) 
                   + sum(np.log(self.P[:,m])*X_test[i,:]+
                   (1 - X_test[i,:])*np.log(1-self.P[:,m])) 
                   for m in range(self.M)]))
        return

"""
APPLY TO TEST SET

test_loglike_list = {}
n_test = test_x.shape[0]

#Here, we first calculate the log-likelihood of an observation under 
# a particular digit model, for each value of M
test_class_list = {}
for M in M_list:
    test_class_list[M] = np.zeros((n_test))
    test_loglike_list[M] = np.zeros((n_test, 10))
    for k in k_list:
        results_list[M][k].predicted_loglik(test_x)
        test_loglike_list[M][:,k] = results_list[M][k].predict_loglik
    
    #Next, we predict a class to each test observation 
    #based on which class has the highest log-likelihood
    test_class_list[M] = [np.argmax(test_loglike_list[M][i,:]) 
                          for i in range(n_test)]
    
error = np.zeros((3))

#Print the accuracy rate for each M
for position in range(3):
    error[position]=sum(test_class_list[M_list[position]]!=test_y)/n_test
    print("Accuracy Rate of M = ", 
          M_list[position], " is ", 1 - error[position])
