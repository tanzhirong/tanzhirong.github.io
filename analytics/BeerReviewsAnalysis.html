Beer Review Analysis via Optimization¶

In this project, I work with 2,924,163 beer reviews from https://www.ratebeer.com. One entry contains the text of the consumer’s review together with ratings of appearance, aroma, palate, style, taste, and an overall score from 0 to 20. The data also include the name, id and brewer of the beer.

To simplify the response, I will give reviews binary label: 1 for a (positive) review with overall score at least 14, and otherwise 0 for a (negative) review.

PART 2¶

Now, our first task is to classify the sentiment of reviews from the text. Text can have neutral, mixed, sarcastic, or otherwise ambiguous sentiment.

We will start to represent text reviews in terms of a vector of features (covariates). So, each review would be a p-dimensional vector where p is the total size of our dictionary. If a review contains a word, that corresponding space = 1, and the rest are set to 0. To ensure algorithmic efficiency, we reduce our matrix into a sparse matrix.

In this section, we will clean and transform the reviews into a sparse matrix.

def clean_and_split(s):
    s = s.lower().replace('-',' ').translate(s.maketrans(" ",
    " ",string.punctuation))
    # replace \r\n

    s = re.sub(' +',' ',s.strip())
    return s.split(' ')

#Function to obtain n x p matrix
def obtain_matrix(texts):
    dictionary = corpora.Dictionary([vocab.keys()])
    corpus = [dictionary.doc2bow(text) for text in texts]
    n1 = len(texts)
    p = len(vocab)
    csr_row=[]
    csr_col=[]
    csr_data=[]
    row_id = 0
    for item in corpus:
        for word_id,count in item:
            csr_row.append(row_id)
            csr_col.append(word_id)
            csr_data.append(count)
        row_id += 1
    sparse_mat = sparse.csr_matrix((csr_data,
                                    (csr_row,csr_col)),shape=(n1,p))
    return sparse_mat

def  cal_cost(theta, X, Y, lamb):
    '''
    Calculates the cost for given X and Y.
    theta = Vector of thetas     
    where j is the number of features
    '''
    firstpart = sum(np.logaddexp(0,X @ theta))
    secondpart = (Y.T @ X @ theta)
    penalty = lamb*np.linalg.norm(theta)*np.linalg.norm(theta)
    return firstpart - secondpart + penalty

    
def cal_error(theta, X, Y):
    '''
    Calculates the prediction error given X and Y.
    theta = Vector of thetas
    '''
    eta = X.dot(theta)
    
    #Calculate pi
    pi = np.exp(eta) / (1 + np.exp(eta))
    prediction = np.zeros(len(Y))
    
    #Rule: if pi > 0.5, then classify 1, o.w. 0
    for i in range(len(Y)):
        if pi[i] >= 0.5:
            prediction[i] = 1
    error = sum(Y != prediction) / len(Y)
    return error

Constant Learning Rate

def mini_gradient_descent_constant(X, Y, X_test, Y_test, theta, lamb,
                          learning_rate = 10,
                          iterations=200,
                          batch_size = 32):
    '''
    X    = Matrix of X
    Y    = Vector of Y
    X_test + Y_test = validation set
    theta = Vector of thetas
    learning_rate  = constant
    iterations = number of epoches
    
    Returns the final theta vector, and array of cost history, and
    array of errors over no of iterations for BOTH training and
    validation set
    '''
    n = len(Y)
    
    cost_history = [0]
    valid_cost_history = [0]
    error_history = [0]
    valid_error_history = [0]
    difference = 100000000
    iteration = 0
    
    for it in range(iterations):
        iteration += 1
        cost = 0.0
        valid_cost = 0.0
        indices = np.random.permutation(n)
        X = X[indices]
        Y = np.array(Y)[indices]
        for i in range(0, n, batch_size):
            X_i = X[i:i+batch_size]
            Y_i = Y[i:i+batch_size]
            #Y_i = np.ndarray.tolist(Y_i)
            
            full_X_i = X_i
           
            #Derivative of Loss
            firstpart = np.array(Y_i*full_X_i)
            firstpart = firstpart.reshape((len(X_i[0,:],1))
            
            eta_batch = full_X_i.dot(theta)
            exp_term = np.exp(eta_batch)/(1 + np.exp(eta_batch))
            secondpart = full_X_i.T.dot(exp_term)
            
            thirdpart = 2*lamb*theta
            
            loss_diff = -firstpart + secondpart + thirdpart
            theta = theta -(1/batch_size)*learning_rate*loss_diff
            
            cost += cal_cost(theta, X_i, Y_i, lamb)

            
        cost_history.append(cost.astype(float))
        difference = abs(cost_history[-1] - cost_history[-2])
        
        valid_cost = cal_cost(theta, X_test, Y_test, lamb)
        valid_cost_history.append(valid_cost)
        error_train = cal_error(theta, X, Y)
        error_history.append(error_train)
        error_valid = cal_error(theta, X_test, Y_test)
        valid_error_history.append(error_valid)
        
        if iteration %5 == 0:
            print(iteration, difference, cost)
            print(error_valid)

    return theta, cost_history, 
           valid_cost_history, error_history, valid_error_history

In the plot above, we plot the cost of each iteration/epoch. Notice that for a constant learning rate, there may not be convergence, since the objective value is oscillating. Hence, we will improve upon our function by allowing learning rate = $\frac{k}{t}$, where k is an input constant, and t is the number of iterations. The more iterations we go through, the smaller the steps will be, to allow for convergence.

"""
Define new SGD function, where the learning rate
is a function of epoch.
"""

def mini_gradient_descent(X, Y, X_test, Y_test, theta, lamb,
                          learning_constant = 5,
                          threshold = 1,
                          batch_size = 100):
    '''
    X    = Matrix of X
    Y    = Vector of labels
    X_test + Y_test = validation set
    theta = Vector of thetas
    learning_constant  = k
    lamb = lambda selected from tuning
    Threshold = minimum difference between cost before convergence
    
    Returns the final theta vector, and array of cost history, and
    array of errors over no. of iterations for both training and 
    validation
    Iteration will stop upon convergence, defined as
        difference between cost < threshold = 1
    '''
    n = len(Y)
    
    cost_history = [0]
    valid_cost_history = [0]
    error_history = [0]
    valid_error_history = [0]
    difference = 100000000
    iteration = 0
    
    while difference > threshold:
        iteration += 1
        learning_rate = learning_constant/iteration
        cost = 0.0
        valid_cost = 0.0
        indices = np.random.permutation(n)
        X = X[indices]
        Y = np.array(Y)[indices]
        for i in range(0, n, batch_size):
            X_i = X[i:i+batch_size]
            Y_i = Y[i:i+batch_size]
            
            full_X_i = X_i
           
            #Derivative of Loss
            firstpart = np.array(Y_i).dot(full_X_i)
            firstpart = firstpart.reshape((len(X_i[0,:])),1)
            eta_batch = full_X_i.dot(theta)
            exp_term = np.exp(eta_batch)/(1 + np.exp(eta_batch))
            secondpart = full_X_i.T.dot(exp_term)
            thirdpart = 2*lamb*theta
            loss_diff = -firstpart + secondpart + thirdpart
                                          
            #Update theta
            theta = theta -(1/batch_size)*learning_rate*loss_diff
                                          
            #Update cost
            cost += cal_cost(theta, X_i, Y_i, lamb)

            
        cost_history.append(cost)
        #Calculate new difference
        difference = abs(cost_history[-1] - cost_history[-2])
        
        #Record all the values we want to measure
        valid_cost = cal_cost(theta, X_test, Y_test, lamb)
        valid_cost_history.append(valid_cost)
        error_train = cal_error(theta, X, Y)
        error_history.append(error_train)
        error_valid = cal_error(theta, X_test, Y_test)
        valid_error_history.append(error_valid)
        
        #Print info to keep track
        if iteration %5 == 0:
            print(iteration, difference)
            print(error_valid)

    return theta, cost_history, valid_cost_history, 
           error_history, valid_error_history
